# Gold Layer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Compute technical indicators (RSI, MACD, Bollinger Bands, ATR, volume ratio, momentum) per asset and aggregated sentiment features (rolling windows, post-volume spike) from the Silver layer, writing the merged result to BigQuery `gold.ml_features`, running in the same Cloud Run Job as ingestion and Silver.

**Architecture:** New `gold/` Python package (mirrors `silver/`) reads the full `silver.ohlcv`, `silver.reddit_posts`, and `silver.crypto_news` tables via the existing `read_table()`, computes indicators and sentiment aggregates, merges them, anti-joins against existing `gold.ml_features` keys, and appends only new rows. `pipeline.py` gains a third stage after ingestion and Silver. No label column — deferred to Plan 5, which will join Gold features against `silver.ohlcv` at training time.

**Tech Stack:** Python 3.11, pandas, `pandas-ta-classic` (new dependency — see Task 1 for why), existing `adapters.WarehouseAdapter`, pytest, ruff, Docker, Cloud Run Jobs, Cloud Scheduler (unchanged), BigQuery.

**Reference spec:** `docs/superpowers/specs/2026-06-29-gold-layer-design.md`

Create `gold/indicators.py`:

```python
import pandas as pd
import pandas_ta_classic as ta

_MOMENTUM_PERIODS = (1, 3, 6)
_VOLUME_RATIO_WINDOW = 20
_INDICATOR_COLS = [
    "rsi_14", "macd_line", "macd_signal", "macd_hist",
    "bb_upper", "bb_lower", "bb_width", "atr_14", "volume_ratio",
    "momentum_1", "momentum_3", "momentum_6",
]


def compute_indicators(ohlcv_df: pd.DataFrame) -> pd.DataFrame:
    if ohlcv_df.empty:
        return ohlcv_df

    results = []
    for _, group in ohlcv_df.groupby("asset"):
        df = group.sort_values("open_time").reset_index(drop=True)

        rsi = ta.rsi(df["close"], length=14)
        macd = ta.macd(df["close"], fast=12, slow=26, signal=9)
        bb = ta.bbands(df["close"], length=20, std=2)
        atr = ta.atr(df["high"], df["low"], df["close"], length=14)

        # pandas_ta_classic returns None (not a NaN-filled Series/DataFrame)
        # when there isn't enough history to compute an indicator at all.
        df["rsi_14"] = rsi if rsi is not None else float("nan")
        df["macd_line"] = macd["MACD_12_26_9"] if macd is not None else float("nan")
        df["macd_signal"] = macd["MACDs_12_26_9"] if macd is not None else float("nan")
        df["macd_hist"] = macd["MACDh_12_26_9"] if macd is not None else float("nan")
        df["bb_upper"] = bb["BBU_20_2.0"] if bb is not None else float("nan")
        df["bb_lower"] = bb["BBL_20_2.0"] if bb is not None else float("nan")
        df["bb_width"] = bb["BBB_20_2.0"] if bb is not None else float("nan")
        df["atr_14"] = atr if atr is not None else float("nan")
        df["volume_ratio"] = df["volume"] / df["volume"].rolling(_VOLUME_RATIO_WINDOW).mean()

        for period in _MOMENTUM_PERIODS:
            df[f"momentum_{period}"] = df["close"].pct_change(periods=period)

        results.append(df)

    combined = pd.concat(results, ignore_index=True)
    combined = combined.dropna(subset=_INDICATOR_COLS).reset_index(drop=True)
    return combined[["asset", "open_time"] + _INDICATOR_COLS]
```

`open`, `high`, `low`, `close`, `volume` are needed as inputs to compute the indicators above but aren't part of the Gold schema (spec section 5) — the final column selection drops them so they don't leak into `gold.ml_features`.

```python
import pandas as pd

_WINDOWS = {
    "sentiment_4h": pd.Timedelta(hours=4),
    "sentiment_24h": pd.Timedelta(hours=24),
    "sentiment_72h": pd.Timedelta(hours=72),
}
_NEWS_WINDOW = pd.Timedelta(hours=24)
_SPIKE_WINDOW = pd.Timedelta(hours=4)
_SPIKE_LOOKBACK = pd.Timedelta(days=7)
_SPIKE_CANDLES = 42  # 7 days of 4h candles

_POST_COLUMNS = ["published_at", "sentiment", "confidence"]


def _normalize(posts: pd.DataFrame) -> pd.DataFrame:
    if posts.empty:
        return pd.DataFrame(columns=_POST_COLUMNS)
    posts = posts.copy()
    posts["published_at"] = pd.to_datetime(posts["published_at"], utc=True)
    return posts


def _weighted_sentiment(posts: pd.DataFrame, end: pd.Timestamp, window: pd.Timedelta) -> float:
    in_window = posts[
        (posts["published_at"] > end - window) & (posts["published_at"] <= end)
    ]
    if in_window.empty:
        return 0.0
    weights = in_window["confidence"]
    if weights.sum() == 0:
        return 0.0
    return float((in_window["sentiment"] * weights).sum() / weights.sum())


def _post_volume_spike(posts: pd.DataFrame, open_time: pd.Timestamp) -> bool:
    if posts.empty:
        return False
    earliest = posts["published_at"].min()
    if earliest > open_time - _SPIKE_LOOKBACK:
        return False

    current_count = len(posts[
        (posts["published_at"] > open_time - _SPIKE_WINDOW)
        & (posts["published_at"] <= open_time)
    ])
    lookback_count = len(posts[
        (posts["published_at"] > open_time - _SPIKE_LOOKBACK)
        & (posts["published_at"] <= open_time)
    ])
    avg_per_candle = lookback_count / _SPIKE_CANDLES
    if avg_per_candle == 0:
        return False
    return bool(current_count > 2 * avg_per_candle)


def compute_sentiment_features(
    reddit_df: pd.DataFrame, news_df: pd.DataFrame, candle_times: pd.Series
) -> pd.DataFrame:
    reddit_df = _normalize(reddit_df)
    news_df = _normalize(news_df)
    combined_df = pd.concat([reddit_df, news_df], ignore_index=True)

    rows = []
    for open_time in sorted(set(candle_times)):
        row = {"open_time": open_time}
        for col, window in _WINDOWS.items():
            row[col] = _weighted_sentiment(combined_df, open_time, window)
        row["news_sentiment_24h"] = _weighted_sentiment(news_df, open_time, _NEWS_WINDOW)
        row["post_volume_spike"] = _post_volume_spike(combined_df, open_time)
        rows.append(row)

    return pd.DataFrame(rows)
```

```python
import pandas as pd

from adapters import get_warehouse
from gold.indicators import compute_indicators
from gold.sentiment_features import compute_sentiment_features


def compute_and_write_features(warehouse) -> None:
    ohlcv_df = warehouse.read_table("silver", "ohlcv")
    reddit_df = warehouse.read_table("silver", "reddit_posts")
    news_df = warehouse.read_table("silver", "crypto_news")
    gold_df = warehouse.read_table("gold", "ml_features")

    indicators_df = compute_indicators(ohlcv_df)
    if indicators_df.empty:
        print("[gold] no new rows")
        return

    sentiment_df = compute_sentiment_features(reddit_df, news_df, indicators_df["open_time"])
    merged = indicators_df.merge(sentiment_df, on="open_time", how="left")

    existing_keys = set(zip(gold_df.get("asset", []), gold_df.get("open_time", [])))
    keys = list(zip(merged["asset"], merged["open_time"]))
    merged = merged[[k not in existing_keys for k in keys]].reset_index(drop=True)

    if merged.empty:
        print("[gold] no new rows")
        return

    merged["computed_at"] = pd.Timestamp.now(tz="UTC")
    warehouse.write_table(merged, "gold", "ml_features", mode="append")
    print(f"[gold] wrote {len(merged)} row(s)")


def run_gold_cycle(warehouse) -> None:
    compute_and_write_features(warehouse)


if __name__ == "__main__":
    wh = get_warehouse()
    run_gold_cycle(wh)
    print("Gold cycle complete")
```
